// IntelliSip ESP32 Client #include <WiFi.h> #include <HTTPClient.h> #include <ArduinoJson.h> const char* ssid = "YOUR_WIFI"; const char* password = "YOUR_PASS"; const char* server = "http://your-server/api"; void sendSip(float water_ml, float temp) { HTTPClient http; http.begin(String(server) + "/ingest-sip"); http.addHeader("Content-Type", "application/json"); StaticJsonDocument<200> doc; doc["user_id"] = "U01"; doc["timestamp"] = getTimestamp(); doc["water_ml"] = water_ml; doc["temperature"] = temp; String body; serializeJson(doc, body); int code = http.POST(body); http.end(); }
